ggplot绘图篇

2023-12-14

ggplot(data = mtcars,aes(x=wt,y=mpg,color=am))+
  geom_point()

ggplot(data = mtcars,aes(x=wt,y=mpg,shape=am,color=am))+
  geom_point() #shape改性状，color颜色标记

#曲线拟合
ggplot(data = mtcars,aes(x=wt,y=mpg,color=am))+geom_smooth()
## `geom_smooth()` using method = 'loess' and formula = 'y ~ x'

#geom_smooth里的参数method默认为loess,即局部加权回归，可以改变参数，想用直线回归，可以将method改为lm
ggplot(data = mtcars,aes(x=wt,y=mpg,color=am))+geom_smooth(method = "lm")
## `geom_smooth()` using formula = 'y ~ x'

#添加散点
ggplot(data = mtcars,aes(x=wt,y=mpg,color=am))+
  geom_point(aes(color=am))+
  stat_smooth()
## `geom_smooth()` using method = 'loess' and formula = 'y ~ x'
